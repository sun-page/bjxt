第二章基础部分
库操作
1
新建数据库
##    create database [if not exists] '数据库名';
2
选择数据库
##    use '数据库名'
登录时直接选择mysql -u root -p ****  '数据库名';
3
查看数据库
##    show databases;
4
查看指定数据库
##    show database '数据库名';
5
删除数据库
##    drop database [if exists] '数据库名';
表操作
6
创建数据表
##    mysql -h localhost -u root -p ****;
7
创建数据表
##     
            create table [if not exists] '表名'(
                      '字段名'  '字段类型'  表的约束,
                      '字段名'  '字段类型'  表的约束,
                    	...
                    );
8
查看所有表
##    show tables;
9
数据表信息
##    show table status [from 数据库名] [like 匹配] \G;
10
修改表名称
##    alter table 旧表名 rename to|as 新表名; 
       rename table 旧表名 to 新表名 [,旧表名 to 新表名]...;
11
修改表字符集
##    alter table 表名 表选项 [=] 值;
12
查看表结构
##    desc|describe '数据表名'; 
       desc|describe '数据表名' '字段名';
13
查看表创建语句
##    show create table '数据表名';
14
修改字段名
##    alter table 数据表名 change 旧字段名 新字段名 字段类型 [字段属性];
15
修改字段类型
##    alter table 数据表名 modify 字段名 新类型 [字段属性];
16
修改字段位置
##    alter table 数据表名 modify 字段名1 数据类型 [字段属性] [first|after 字段名2];
17
新增字段
##    alter table 数据表名 add 新字段名 字段类型 [first|after 原有字段名]; 
       alter table 数据表名 add (新字段名1 字段类型1,新字段名2 字段类型2,... );
18
删除字段
##    alter table 数据表名 drop 字段名;
19
删除数据表
##    drop table [if exists] 数据表名 ;
数据操作
20
添加数据
##    insert [into] 数据表名 values(值1,值2,...); 
       insert [into] 数据表名 [(字段名1,字段名2,...)] values (值1,值2,...);
       insert [into] 数据表名 [(字段名1,字段名2,...)] values (值1,值2,...),(值1,值2,...);
21
查询数据
##    select * from 数据表名; 
       select {字段名1,字段名,..} from 数据表名;
22
条件查询数据
##    select *|{字段名 1, 字段名 2 ,..} from 数据表名 where 字段名=值;
23
修改数据
##    update 数据表名 set 字段名 1 = 值 1 [,字段名 2 = 值 2] [where 条件];
24
删除数据
##    delete from 数据表名 [where 条件]; 
删除表所有数据delete from 数据表名
